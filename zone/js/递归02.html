	let saveClimb = []
	function climbStairs(n){
		if (n === 1) return 1
		if (n === 2) return 2
		if( !saveClimb[n-1] ){
			saveClimb[n-1] = climbStairs(n-1)
		}
		if( !saveClimb[n-2] ){
			saveClimb[n-2] = climbStairs(n-2)
		}
		return saveClimb[n-1] + saveClimb[n-2]
	}
